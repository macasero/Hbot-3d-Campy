en \sys\config-var.g

; ==============================================================
;
; Hbot XL by Campy3d
;
global model = 2		; 1 = XL 		2 = ST    for  CUSTOM (edit in config-var.g on you model type XT or ST)
global probetype = 2		; 1 = IR mini	2 = Superpinda
; ===============================================================
